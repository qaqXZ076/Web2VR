   372тЖТ   372тЖТ    // the 2D loop. This guards against a race where onSessionEnded fires but
   373тЖТ   373тЖТ    // the XR runtime hasn't fully released the GL context yet.
   374тЖТ   374тЖТ    if (renderer.xr.isPresenting) {

   489тЖТ   489тЖТ  // ====== onSessionEnded: Standard WebXR session cleanup ======
   490тЖТ   490тЖТ  // This function handles ALL cleanup when an XRSession ends тАФ whether
   491тЖТ   491тЖТ  // triggered by the user clicking "Exit VR", taking off the headset, or
   492тЖТ   492тЖТ  // SteamVR forcing a disconnect. It follows the standard WebXR lifecycle:
   493тЖТ   493тЖТ  //
   494тЖТ   494тЖТ  // 1. Stop the Three.js render loop (release the XR frame pipeline)
   495тЖТ   495тЖТ  // 2. Clear the session reference so no stale reference blocks re-entry
   496тЖТ   496тЖТ  // 3. Reset app state (isInVR, showPreview, camera layers)
   497тЖТ   497тЖТ  // 4. Restart the render loop for non-VR (2D/flat) rendering
   498тЖТ   498тЖТ  //
   499тЖТ   499тЖТ  // IMPORTANT: This is registered as a listener on the XRSession's 'end'
   500тЖТ   500тЖТ  // event. It MUST be called for every session termination path. The 'end'
   501тЖТ   501тЖТ  // event fires after session.end() resolves, which means the browser has
   502тЖТ   502тЖТ  // successfully sent the disconnect signal to SteamVR/OpenXR.
   503тЖТ   503тЖТ  const onSessionEnded = useCallback(() => {
   504тЖТ   504тЖТ    console.log('[VR] onSessionEnded: VR Session has ended, cleaning up...');
   505тЖТ   505тЖТ    const renderer = rendererRef.current;
   506тЖТ   506тЖТ    const camera = cameraRef.current;
   507тЖТ   507тЖТ
   508тЖТ   508тЖТ    // 1. Stop the render loop тАФ this releases the XR frame submission pipeline.
   509тЖТ   509тЖТ    // Without this, the loop keeps submitting frames to a dead session,
   510тЖТ   510тЖТ    // which keeps SteamVR connected to Chrome.
   511тЖТ   511тЖТ    if (renderer) {
   512тЖТ   512тЖТ      renderer.setAnimationLoop(null);
   513тЖТ   513тЖТ    }
   514тЖТ   514тЖТ
   515тЖТ   515тЖТ    // 2. Clear our session reference
   516тЖТ   516тЖТ    vrSessionRef.current = null;
   517тЖТ   517тЖТ
   518тЖТ   518тЖТ    // 3. Reset app state
   519тЖТ   519тЖТ    setIsInVR(false);
   520тЖТ   520тЖТ    setShowPreview(true);
   521тЖТ   521тЖТ    if (camera) {
   522тЖТ   522тЖТ      camera.layers.enable(1);
   523тЖТ   523тЖТ      camera.layers.enable(2);
   524тЖТ   524тЖТ    }
   525тЖТ   525тЖТ
   526тЖТ   526тЖТ    // 4. Restart the render loop for non-VR (2D/flat) rendering.
   527тЖТ   527тЖТ    // Must use setTimeout to ensure Three.js has fully processed the session
   528тЖТ   528тЖТ    // end before we restart the loop. Without the delay, the new animation
   529тЖТ   529тЖТ    // loop might try to use the old XR context.
   530тЖТ   530тЖТ    // startRenderLoop also explicitly verifies renderer.xr.isPresenting is
   531тЖТ   531тЖТ    // false and resets renderer.xr.enabled as a safety measure.
   532тЖТ   532тЖТ    setTimeout(() => {
   533тЖТ   533тЖТ      startRenderLoop();
   534тЖТ   534тЖТ    }, 100);
   535тЖТ   535тЖТ  }, [setIsInVR, startRenderLoop]);
   536тЖТ   536тЖТ
   537тЖТ   537тЖТ  // ====== Enter VR ======
   538тЖТ   538тЖТ  const handleEnterVR = useCallback(async () => {
   539тЖТ   539тЖТ    const renderer = rendererRef.current;
   540тЖТ   540тЖТ    const camera = cameraRef.current;
   541тЖТ   541тЖТ    if (!renderer || !navigator.xr || !camera) return;
   542тЖТ   542тЖТ
   543тЖТ   543тЖТ    // Debounce: prevent rapid clicks from creating multiple sessions
   544тЖТ   544тЖТ    if (vrEnteringRef.current) {
   545тЖТ   545тЖТ      console.log('[VR] Already entering VR, ignoring click');
   546тЖТ   546тЖТ      return;
   547тЖТ   547тЖТ    }
   548тЖТ   548тЖТ
   549тЖТ   549тЖТ    try {
   550тЖТ   550тЖТ      vrEnteringRef.current = true;
   551тЖТ   551тЖТ
   552тЖТ   552тЖТ      // Check if there's an existing active session via renderer.xr.getSession().
   553тЖТ   553тЖТ      // This is more reliable than vrSessionRef because it queries Three.js's
   554тЖТ   554тЖТ      // actual state directly, avoiding reference desync issues.
   555тЖТ   555тЖТ      const currentSession = renderer.xr.getSession();
   556тЖТ   556тЖТ      if (currentSession) {
   557тЖТ   557тЖТ        console.warn('[VR] Existing session found during enterVR, ending it first');
   558тЖТ   558тЖТ        // End the existing session and wait for it to fully terminate.
   559тЖТ   559тЖТ        // This ensures SteamVR/OpenXR properly disconnects before we request
   560тЖТ   560тЖТ        // a new session.
   561тЖТ   561тЖТ        try {
   562тЖТ   562тЖТ          await currentSession.end();
   563тЖТ   563тЖТ        } catch { /* may already be ending */ }
   564тЖТ   564тЖТ        // Give the runtime a moment to fully process the session end
   565тЖТ   565тЖТ        await new Promise<void>((resolve) => setTimeout(resolve, 200));
   566тЖТ   566тЖТ      }
   567тЖТ   567тЖТ
   568тЖТ   568тЖТ      // Request a new immersive-vr session from the browser
   569тЖТ   569тЖТ      const session = await navigator.xr.requestSession('immersive-vr', {
   570тЖТ   570тЖТ        optionalFeatures: ['local-floor', 'bounded-floor'],
   571тЖТ   571тЖТ      });
   572тЖТ   572тЖТ      vrSessionRef.current = session;
   573тЖТ   573тЖТ
   574тЖТ   574тЖТ      // CRITICAL: Register the 'end' event listener on the session BEFORE
   575тЖТ   575тЖТ      // anything else. This ensures that no matter how the session ends тАФ
   576тЖТ   576тЖТ      // user clicks Exit VR, takes off headset, SteamVR force-disconnects тАФ
   577тЖТ   577тЖТ      // our cleanup handler runs and properly terminates the XR pipeline.
   578тЖТ   578тЖТ      //
   579тЖТ   579тЖТ      // Per WebXR standard: session.end() is async. The 'end' event fires
   580тЖТ   580тЖТ      // after the browser has successfully sent the disconnect signal to
   581тЖТ   581тЖТ      // the XR runtime (SteamVR/OpenXR). All cleanup must happen here.
   582тЖТ   582тЖТ      session.addEventListener('end', onSessionEnded);
   583тЖТ   583тЖТ
   584тЖТ   584тЖТ      // Pass the session to Three.js's XR manager
   585тЖТ   585тЖТ      await renderer.xr.setSession(session);
   586тЖТ   586тЖТ
   587тЖТ   587тЖТ      // Restart the animation loop with the new XR session context.
   588тЖТ   588тЖТ      // After renderer.xr.setSession(), the animation loop uses the
   589тЖТ   589тЖТ      // session's requestAnimationFrame for VR frame timing.
   590тЖТ   590тЖТ      startRenderLoop();
   591тЖТ   591тЖТ
   592тЖТ   592тЖТ      setIsInVR(true);
   593тЖТ   593тЖТ      setShowPreview(false);
   594тЖТ   594тЖТ

   596тЖТ   596тЖТ      const video = videoRef.current;

   599тЖТ   599тЖТ          buildScene(cropCanvasRef.current);
   600тЖТ   600тЖТ        } else {
   601тЖТ   601тЖТ          buildScene(video);
   602тЖТ   602тЖТ        }
   603тЖТ   603тЖТ      } else if (captureMode === 'test' && sceneRef.current && meshesRef.current.length > 0) {
   604тЖТ   604тЖТ        // Test pattern: meshes are already in the scene, just make sure they're visible
   605тЖТ   605тЖТ      }
   606тЖТ   606тЖТ    } catch (err) {
   607тЖТ   607тЖТ      console.error('Failed to start VR session:', err);
   608тЖТ   608тЖТ      setVideoError(
   609тЖТ   609тЖТ        `Failed to enter VR: ${err instanceof Error ? err.message : 'Unknown error'}. Make sure your VR headset is connected and SteamVR is running.`
   610тЖТ   610тЖТ      );
   611тЖТ   611тЖТ    } finally {
   612тЖТ   612тЖТ      vrEnteringRef.current = false;
   613тЖТ   613тЖТ    }
   614тЖТ   614тЖТ  }, [setIsInVR, buildScene, cropEnabled, storeCropRegion, captureMode, startRenderLoop, onSessionEnded]);

   616тЖТ   616тЖТ  // ====== Exit VR ======

   619тЖТ   619тЖТ    // active session. This is more reliable than vrSessionRef because it

  1223тЖТ  1223тЖТ// ====== Create VR Meshes ======
  1224тЖТ  1224тЖТ
  1225тЖТ  1225тЖТfunction createVRMeshes(
  1226тЖТ  1226тЖТ  texture: THREE.Texture,
  1227тЖТ  1227тЖТ  layout: StereoLayout,
  1228тЖТ  1228тЖТ  projection: ProjectionType,
  1229тЖТ  1229тЖТ  fov: number
  1230тЖТ  1230тЖТ): THREE.Mesh[] {
  1231тЖТ  1231тЖТ  const meshes: THREE.Mesh[] = [];
  1232тЖТ  1232тЖТ
  1233тЖТ  1233тЖТ  function createGeometry(phiLength: number): THREE.BufferGeometry {
  1234тЖТ  1234тЖТ    switch (projection) {
  1235тЖТ  1235тЖТ      case 'sphere360':
  1236тЖТ  1236тЖТ        return new THREE.SphereGeometry(500, 80, 40, 0, phiLength);
  1237тЖТ  1237тЖТ      case 'hemisphere180':
  1238тЖТ  1238тЖТ        return new THREE.SphereGeometry(500, 80, 40, 0, phiLength);
  1239тЖТ  1239тЖТ      case 'cylinder': {
  1240тЖТ  1240тЖТ        const fovRad = (fov * Math.PI) / 180;
  1241тЖТ  1241тЖТ        const radius = 500;
  1242тЖТ  1242тЖТ        return new THREE.CylinderGeometry(radius, radius, radius * 0.5625, 64, 1, true, -fovRad / 2 + Math.PI / 2, fovRad);
  1243тЖТ  1243тЖТ      }
  1244тЖТ  1244тЖТ      default:
  1245тЖТ  1245тЖТ        return new THREE.SphereGeometry(500, 80, 40, 0, phiLength);
  1246тЖТ  1246тЖТ    }
  1247тЖТ  1247тЖТ  }
  1248тЖТ  1248тЖТ
  1249тЖТ  1249тЖТ  function modifyUVs(geometry: THREE.BufferGeometry, region: { x: number; y: number; w: number; h: number }): THREE.BufferGeometry {
  1250тЖТ  1250тЖТ    const uvAttr = geometry.getAttribute('uv');
  1251тЖТ  1251тЖТ    if (!uvAttr) return geometry;
  1252тЖТ  1252тЖТ    const uvArray = uvAttr.array as Float32Array;
  1253тЖТ  1253тЖТ    for (let i = 0; i < uvArray.length; i += 2) {
  1254тЖТ  1254тЖТ      uvArray[i] = region.x + uvArray[i] * region.w;
  1255тЖТ  1255тЖТ      uvArray[i + 1] = region.y + uvArray[i + 1] * region.h;
  1256тЖТ  1256тЖТ    }
  1257тЖТ  1257тЖТ    uvAttr.needsUpdate = true;
  1258тЖТ  1258тЖТ    return geometry;
  1259тЖТ  1259тЖТ  }
  1260тЖТ  1260тЖТ
  1261тЖТ  1261тЖТ  const phiLength =
  1262тЖТ  1262тЖТ    projection === 'sphere360' ? 2 * Math.PI :
  1263тЖТ  1263тЖТ    projection === 'hemisphere180' ? Math.PI :
  1264тЖТ  1264тЖТ    (fov * Math.PI) / 180;
  1265тЖТ  1265тЖТ
  1266тЖТ  1266тЖТ  switch (layout) {
  1267тЖТ  1267тЖТ    case 'sbs': {
  1268тЖТ  1268тЖТ      const leftGeom = createGeometry(phiLength);
  1269тЖТ  1269тЖТ      modifyUVs(leftGeom, { x: 0, y: 0, w: 0.5, h: 1 });
  1270тЖТ  1270тЖТ      const leftMesh = new THREE.Mesh(leftGeom, new THREE.MeshBasicMaterial({ map: texture, side: THREE.BackSide }));
  1271тЖТ  1271тЖТ      leftMesh.layers.set(1);
  1272тЖТ  1272тЖТ      meshes.push(leftMesh);
  1273тЖТ  1273тЖТ
  1274тЖТ  1274тЖТ      const rightGeom = createGeometry(phiLength);
  1275тЖТ  1275тЖТ      modifyUVs(rightGeom, { x: 0.5, y: 0, w: 0.5, h: 1 });
  1276тЖТ  1276тЖТ      const rightMesh = new THREE.Mesh(rightGeom, new THREE.MeshBasicMaterial({ map: texture, side: THREE.BackSide }));
  1277тЖТ  1277тЖТ      rightMesh.layers.set(2);
  1278тЖТ  1278тЖТ      meshes.push(rightMesh);
  1279тЖТ  1279тЖТ      break;
  1280тЖТ  1280тЖТ    }
  1281тЖТ  1281тЖТ    case 'tb': {
  1282тЖТ  1282тЖТ      const leftGeom = createGeometry(phiLength);
  1283тЖТ  1283тЖТ      modifyUVs(leftGeom, { x: 0, y: 0.5, w: 1, h: 0.5 });
  1284тЖТ  1284тЖТ      const leftMesh = new THREE.Mesh(leftGeom, new THREE.MeshBasicMaterial({ map: texture, side: THREE.BackSide }));
  1285тЖТ  1285тЖТ      leftMesh.layers.set(1);
  1286тЖТ  1286тЖТ      meshes.push(leftMesh);
  1287тЖТ  1287тЖТ
  1288тЖТ  1288тЖТ      const rightGeom = createGeometry(phiLength);
  1289тЖТ  1289тЖТ      modifyUVs(rightGeom, { x: 0, y: 0, w: 1, h: 0.5 });
  1290тЖТ  1290тЖТ      const rightMesh = new THREE.Mesh(rightGeom, new THREE.MeshBasicMaterial({ map: texture, side: THREE.BackSide }));
  1291тЖТ  1291тЖТ      rightMesh.layers.set(2);
  1292тЖТ  1292тЖТ      meshes.push(rightMesh);
  1293тЖТ  1293тЖТ      break;
  1294тЖТ  1294тЖТ    }
  1295тЖТ  1295тЖТ    case 'mono': {
  1296тЖТ  1296тЖТ      const geom = createGeometry(phiLength);
  1297тЖТ  1297тЖТ      const mesh = new THREE.Mesh(geom, new THREE.MeshBasicMaterial({ map: texture, side: THREE.BackSide }));
  1298тЖТ  1298тЖТ      mesh.layers.enable(1);
  1299тЖТ  1299тЖТ      mesh.layers.enable(2);
  1300тЖТ  1300тЖТ      meshes.push(mesh);
  1301тЖТ  1301тЖТ      break;
  1302тЖТ  1302тЖТ    }
  1303тЖТ  1303тЖТ  }
  1304тЖТ  1304тЖТ  return meshes;
  1305тЖТ  1305тЖТ}
