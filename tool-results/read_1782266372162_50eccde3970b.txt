   814тЖТ   814тЖТ// ====== Create VR Meshes ======
   815тЖТ   815тЖТ
   816тЖТ   816тЖТfunction createVRMeshes(
   817тЖТ   817тЖТ  texture: THREE.Texture,
   818тЖТ   818тЖТ  layout: StereoLayout,
   819тЖТ   819тЖТ  projection: ProjectionType,
   820тЖТ   820тЖТ  fov: number
   821тЖТ   821тЖТ): THREE.Mesh[] {
   822тЖТ   822тЖТ  const meshes: THREE.Mesh[] = [];
   823тЖТ   823тЖТ
   824тЖТ   824тЖТ  function createGeometry(phiLength: number): THREE.BufferGeometry {
   825тЖТ   825тЖТ    switch (projection) {
   826тЖТ   826тЖТ      case 'sphere360':
   827тЖТ   827тЖТ        return new THREE.SphereGeometry(500, 80, 40, 0, phiLength);
   828тЖТ   828тЖТ      case 'hemisphere180':
   829тЖТ   829тЖТ        return new THREE.SphereGeometry(500, 80, 40, 0, phiLength);
   830тЖТ   830тЖТ      case 'cylinder': {
   831тЖТ   831тЖТ        const fovRad = (fov * Math.PI) / 180;
   832тЖТ   832тЖТ        const radius = 500;
   833тЖТ   833тЖТ        return new THREE.CylinderGeometry(radius, radius, radius * 0.5625, 64, 1, true, -fovRad / 2 + Math.PI / 2, fovRad);
   834тЖТ   834тЖТ      }
   835тЖТ   835тЖТ      default:
   836тЖТ   836тЖТ        return new THREE.SphereGeometry(500, 80, 40, 0, phiLength);
   837тЖТ   837тЖТ    }
   838тЖТ   838тЖТ  }
   839тЖТ   839тЖТ
   840тЖТ   840тЖТ  function modifyUVs(geometry: THREE.BufferGeometry, region: { x: number; y: number; w: number; h: number }): THREE.BufferGeometry {
   841тЖТ   841тЖТ    const uvAttr = geometry.getAttribute('uv');
   842тЖТ   842тЖТ    if (!uvAttr) return geometry;
   843тЖТ   843тЖТ    const uvArray = uvAttr.array as Float32Array;
   844тЖТ   844тЖТ    for (let i = 0; i < uvArray.length; i += 2) {
   845тЖТ   845тЖТ      uvArray[i] = region.x + uvArray[i] * region.w;
   846тЖТ   846тЖТ      uvArray[i + 1] = region.y + uvArray[i + 1] * region.h;
   847тЖТ   847тЖТ    }
   848тЖТ   848тЖТ    uvAttr.needsUpdate = true;
   849тЖТ   849тЖТ    return geometry;
   850тЖТ   850тЖТ  }
   851тЖТ   851тЖТ
   852тЖТ   852тЖТ  const phiLength =
   853тЖТ   853тЖТ    projection === 'sphere360' ? 2 * Math.PI :
   854тЖТ   854тЖТ    projection === 'hemisphere180' ? Math.PI :
   855тЖТ   855тЖТ    (fov * Math.PI) / 180;
   856тЖТ   856тЖТ
   857тЖТ   857тЖТ  switch (layout) {
   858тЖТ   858тЖТ    case 'sbs': {
   859тЖТ   859тЖТ      const leftGeom = createGeometry(phiLength);
   860тЖТ   860тЖТ      modifyUVs(leftGeom, { x: 0, y: 0, w: 0.5, h: 1 });
   861тЖТ   861тЖТ      const leftMesh = new THREE.Mesh(leftGeom, new THREE.MeshBasicMaterial({ map: texture, side: THREE.BackSide }));
   862тЖТ   862тЖТ      leftMesh.layers.set(1);
   863тЖТ   863тЖТ      meshes.push(leftMesh);
   864тЖТ   864тЖТ
   865тЖТ   865тЖТ      const rightGeom = createGeometry(phiLength);
   866тЖТ   866тЖТ      modifyUVs(rightGeom, { x: 0.5, y: 0, w: 0.5, h: 1 });
   867тЖТ   867тЖТ      const rightMesh = new THREE.Mesh(rightGeom, new THREE.MeshBasicMaterial({ map: texture, side: THREE.BackSide }));
   868тЖТ   868тЖТ      rightMesh.layers.set(2);
   869тЖТ   869тЖТ      meshes.push(rightMesh);
   870тЖТ   870тЖТ      break;
   871тЖТ   871тЖТ    }
   872тЖТ   872тЖТ    case 'tb': {
   873тЖТ   873тЖТ      const leftGeom = createGeometry(phiLength);
   874тЖТ   874тЖТ      modifyUVs(leftGeom, { x: 0, y: 0.5, w: 1, h: 0.5 });
   875тЖТ   875тЖТ      const leftMesh = new THREE.Mesh(leftGeom, new THREE.MeshBasicMaterial({ map: texture, side: THREE.BackSide }));
   876тЖТ   876тЖТ      leftMesh.layers.set(1);
   877тЖТ   877тЖТ      meshes.push(leftMesh);
   878тЖТ   878тЖТ
   879тЖТ   879тЖТ      const rightGeom = createGeometry(phiLength);
   880тЖТ   880тЖТ      modifyUVs(rightGeom, { x: 0, y: 0, w: 1, h: 0.5 });
   881тЖТ   881тЖТ      const rightMesh = new THREE.Mesh(rightGeom, new THREE.MeshBasicMaterial({ map: texture, side: THREE.BackSide }));
   882тЖТ   882тЖТ      rightMesh.layers.set(2);
   883тЖТ   883тЖТ      meshes.push(rightMesh);
   884тЖТ   884тЖТ      break;
   885тЖТ   885тЖТ    }
   886тЖТ   886тЖТ    case 'mono': {
   887тЖТ   887тЖТ      const geom = createGeometry(phiLength);
   888тЖТ   888тЖТ      const mesh = new THREE.Mesh(geom, new THREE.MeshBasicMaterial({ map: texture, side: THREE.BackSide }));
   889тЖТ   889тЖТ      mesh.layers.enable(1);
   890тЖТ   890тЖТ      mesh.layers.enable(2);
   891тЖТ   891тЖТ      meshes.push(mesh);
   892тЖТ   892тЖТ      break;
   893тЖТ   893тЖТ    }
   894тЖТ   894тЖТ  }
   895тЖТ   895тЖТ  return meshes;
   896тЖТ   896тЖТ}
